## 1. Sabitler

```csharp
private const int    YENILEME_ARALIĞI_SANIYE = 30;
private const string KARAKTERLER             = "0123456789";
private const int    MAX_DENEME_BEFORE_RESET = 3;
```

## 2. Challenge Kodu — `KodUretici`

### 2A. Zaman Tabanlı (`KodUret`) — TOTP benzeri

```
zamanIndeksi = floor(UTC_Unix / 30)
hashGirdisi  = "{secret}:{zamanIndeksi}"
hash         = HMAC-SHA256(key=secret, input=hashGirdisi)
kod          = hash[0]%10 + hash[1]%10 + hash[2]%10 + hash[3]%10
```

```csharp
public string KodUret()
{
    var zamanIndeksi = DateTimeOffset.UtcNow.ToUnixTimeSeconds() / YENILEME_ARALIĞI_SANIYE;
    var hashGirdisi  = $"{_gizliAnahtar}:{zamanIndeksi}";
    using var hmac   = new HMACSHA256(Encoding.UTF8.GetBytes(_gizliAnahtar));
    var hash         = hmac.ComputeHash(Encoding.UTF8.GetBytes(hashGirdisi));
    return HashiKodaDonustur(hash, KOD_UZUNLUGU);
}

private string HashiKodaDonustur(byte[] hash, int uzunluk)
{
    var kod = new StringBuilder();
    for (int i = 0; i < uzunluk; i++)
        kod.Append(hash[i] % KARAKTERLER.Length);
    return kod.ToString();
}
```

Zaman penceresi:
```
10:00:00–10:00:29  indeks=56988000  → "4827"
10:00:30–10:00:59  indeks=56988001  → "3091"  (yeni)
10:01:00–10:01:29  indeks=56988002  → "7652"  (yeni)
```

### 2B. Rastgele (`RastgeleKodUret`) — 3 hatalı girişte reset için

```csharp
public string RastgeleKodUret()
{
    var random       = new Random();
    var rastgeleSayi = random.Next(1000000, 9999999);
    var zaman        = DateTimeOffset.UtcNow.Ticks;
    var hashGirdisi  = $"{_gizliAnahtar}:{rastgeleSayi}:{zaman}";
    using var hmac   = new HMACSHA256(Encoding.UTF8.GetBytes(_gizliAnahtar));
    return HashiKodaDonustur(hmac.ComputeHash(Encoding.UTF8.GetBytes(hashGirdisi)), KOD_UZUNLUGU);
}
```

### 2C. Kalan Süre

```csharp
public int KalanSaniyeyiGetir()
{
    var s = DateTimeOffset.UtcNow.ToUnixTimeSeconds();
    return (int)(YENILEME_ARALIĞI_SANIYE - (s % YENILEME_ARALIĞI_SANIYE));
}
```

## 3. Response Kodu — `DogrulamaKoduServisi`

```
birlesmis = "{challenge}:{secret}"
hash      = SHA256(UTF8(birlesmis))
response  = hash[0]%10 + hash[1]%10 + hash[2]%10 + hash[3]%10
```

```csharp
public string YanitUret(string dogrulamaKodu)
{
    var birlesmis    = $"{dogrulamaKodu}:{_gizliAnahtar}";
    using var sha256 = SHA256.Create();
    return HashiKodaDonustur(sha256.ComputeHash(Encoding.UTF8.GetBytes(birlesmis)), YANIT_UZUNLUGU);
}

public bool YanitiDogrula(string dogrulamaKodu, string yanitKodu) =>
    YanitUret(dogrulamaKodu).Equals(yanitKodu, StringComparison.OrdinalIgnoreCase);
```

| | `KodUretici` | `DogrulamaKoduServisi` |
|---|---|---|
| Amaç | Challenge üretir | Response üretir |
| Algoritma | HMAC-SHA256 | SHA256 |
| Girdi | `secret:zamanIndeksi` | `challenge:secret` |
| Zaman bağımlı | Her 30 sn değişir | Challenge sabitse sabit |

## 4. QR Sistemi

**URL:** `https://www.kulumtal.com/kilit.php?challenge=4827`

`kilit.php` challenge'ı alır → SHA256 ile response hesaplar → kullanıcıya gösterir.

```csharp
private void QrKoduOlustur(string challengeCode)
{
    string qrUrl = $"https://www.kulumtal.com/kilit.php?challenge={challengeCode}";

    using var qrGenerator = new QRCodeGenerator();
    var qrCodeData        = qrGenerator.CreateQrCode(qrUrl, QRCodeGenerator.ECCLevel.Q);
    using var qrCode      = new QRCode(qrCodeData);
    var qrBitmap          = qrCode.GetGraphic(10);   // ~290×290 px

    using var ms = new MemoryStream();
    qrBitmap.Save(ms, ImageFormat.Png);
    ms.Seek(0, SeekOrigin.Begin);

    var bitmapImage = new BitmapImage();
    bitmapImage.BeginInit();
    bitmapImage.StreamSource = ms;
    bitmapImage.CacheOption  = BitmapCacheOption.OnLoad;
    bitmapImage.EndInit();
    bitmapImage.Freeze();   // cross-thread güvenli
    qrCodeImage.Source = bitmapImage;
}
```

## 5. Otomatik Yenileme Timer

```csharp
// Her 1 saniyede çalışır
private void CodeRefreshTimer_Tick(object? sender, EventArgs e)
{
    var remaining           = _kodUretici.KalanSaniyeyiGetir();
    progressCodeTimer.Value = (remaining / 30.0) * 100;   // %100→%0

    int currentTimeIndex = (int)(DateTimeOffset.UtcNow.ToUnixTimeSeconds() / 30);
    if (currentTimeIndex != _sonKodGuncellemeSuresi)
    {
        _sonKodGuncellemeSuresi = currentTimeIndex;
        DogrulamaKodunuGuncelle();   // challenge + QR + dialog sync
    }
}
```

## 6. `KodDogrulamaPenceresi` — Dialog

### 3 Hatalı Deneme Kuralı

```csharp
private void KoduDogrula()
{
    var girilenKod    = inputBox.Text.Trim().ToUpper();
    var beklenenYanit = _dogrulamaServisi.YanitUret(_suankiDogrulamaKodu);

    if (girilenKod == beklenenYanit)
    {
        Dogrulandi = true; _hataliDeneme = 0; Close();
    }
    else
    {
        _hataliDeneme++;
        if (_hataliDeneme >= MAX_DENEME_BEFORE_RESET)
        {
            _suankiDogrulamaKodu = _yeniChallengeUret();   // delegate → ana pencere
            _hataliDeneme        = 0;
        }
    }
}
```

```
Deneme 1 → Yanlış → "Yanlış kod! (1/3)"
Deneme 2 → Yanlış → "Yanlış kod! (2/3)"
Deneme 3 → Yanlış → RastgeleKodUret() → QR + challenge yenilenir
```

## 8. Kotlin Eşdeğeri (Mobil)

```kotlin
object ChallengeResponseHelper {
    private const val SECRET = "tahta_ekran_secret_2024"

    fun generateResponse(challengeCode: String): String {
        val hash = MessageDigest.getInstance("SHA-256")
            .digest("$challengeCode:$SECRET".toByteArray(Charsets.UTF_8))
        return (0 until 4).map { (hash[it].toInt() and 0xFF) % 10 }.joinToString("")
    }
}
```

> C#'ta `byte` işaretsiz (0–255) olduğundan `and 0xFF` gerekmez. Kotlin'de zorunludur.

## 9. Güvenlik Notları

| Konu | Durum | Öneri |
|---|---|---|
| Secret | Kodda sabit | `appsettings.json` / env variable |
| Brute-force | 3 deneme → reset | Exponential backoff ekle |
| Replay | 30 sn penceresi | Kullanılan response'ları cache'le |
| Response uzunluğu | 4 hane / 10.000 | 6 haneye çıkar → 1.000.000 |
| `tbResponseCode` | Ekranda görünür | Production'da kaldır |
